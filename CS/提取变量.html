# 提取变量

if ($scope.goodSkill.indexOf('analytics') !== -1) {
	skill.analytics = 5
}

5 这个数字就比较难以理解其含义

最简单有效的方式就是提取变量

const LEVEL_FIVE = 5
if ($scope.goodSkill.indexOf('analytics') !== -1) {
	skill.analytics = LEVEL_FIVE
}
